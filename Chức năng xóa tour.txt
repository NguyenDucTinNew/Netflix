public ActionResult Xoa(int id)
        {
            Tour tour = db.tours.Where(row => row.MaTour == id).FirstOrDefault();
            return View(tour);
        }

[HttpPost]
public ActionResult Xoa(int id, Tour p)
        {
            Tour tour = db.tours.Where(row => row.MaTour == id).FirstOrDefault();
            db.tours.Remove(tour);
            db.SaveChanges();
            return RedirectToAction("Index");
        }
